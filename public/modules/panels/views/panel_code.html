var request = require('request'); request({ method: 'PUT', url: 'https://{{host}}/resources/{{topic}}', headers: { 'Content-Type': 'application/json' }, body: '{\"value\": \"{{value}}\"}' }, function (error, response, body) { console.log('Status:', response.statusCode); console.log('Headers:', JSON.stringify(response.headers)); console.log('Response:', body); });
